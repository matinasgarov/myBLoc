> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Expand scoring from 6 to 7 evidence-based factors, add a per-factor breakdown panel to the result screen, and compile metro ridership + settlement data into static lookup files.

**Architecture:** Static data files (`lib/metro-stations.ts`, `lib/settlements.ts`) are generated once from datasets via Node scripts and shipped with the app. At runtime, `lib/overpass.ts` fetches live OSM data, enriches the `PlacesContext` with metro distance, urban tier, bus stop count, parking count, grocery count, and major road count, then `lib/score.ts` computes a 7-factor score returning `{ score, factors }`. The route merges `factors` into the `AnalysisResult`, and `ResultSheet` renders the factor breakdown panel.

**Tech Stack:** Next.js 16, TypeScript, Overpass API, Groq SDK (llama-3.3-70b-versatile), Jest + ts-jest, Tailwind v4, Leaflet (raw), local CSV/GeoJSON datasets.

```ts
export function haversineMetres(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371000
  const φ1 = (lat1 * Math.PI) / 180
  const φ2 = (lat2 * Math.PI) / 180
  const Δφ = ((lat2 - lat1) * Math.PI) / 180
  const Δλ = ((lng2 - lng1) * Math.PI) / 180
  const a = Math.sin(Δφ / 2) ** 2 + Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ / 2) ** 2
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
}
```

The script reads `datasets/pathways.csv` (station exits, semicolon-delimited, coordinates as integers × 1,000,000) and `datasets/passenger_2025.csv` (daily ridership per station, semicolon-delimited), matches stations by name, computes average daily ridership, and writes `lib/metro-stations.ts`.

**Known name mismatches between the two CSVs:**

| pathways.csv name (col 1) | passenger_2025.csv name |
|---|---|
| `28.05.2026 00:00:00` | `28 May` |
| `Şah İsmayıl Xətai` | `Xətai` |
| `Memar Əcəmi 2` | `Memar Əcəmi-2` |
| `Koroğlu` | `Koroglu` |

- [ ] **Step 1: Create `scripts/build-metro-data.mjs`**

```js
// scripts/build-metro-data.mjs
// Run once: node scripts/build-metro-data.mjs
import { readFileSync, writeFileSync } from 'fs'
import { resolve, dirname } from 'path'
import { fileURLToPath } from 'url'

const __dir = dirname(fileURLToPath(import.meta.url))
const root = resolve(__dir, '..')

// --- Parse pathways.csv ---
const pathwaysRaw = readFileSync(resolve(root, 'datasets/pathways.csv'), 'utf-8')
const pathwaysLines = pathwaysRaw.split('\n').slice(1).filter(l => l.trim())

// Group exits by station AZ name
const exitsByStation = new Map() // nameAz → [{lat, lng}]
for (const line of pathwaysLines) {
  const cols = line.split(';').map(c => c.replace(/^"|"$/g, '').trim())
  const nameAz = cols[0]
  const latInt = parseInt(cols[2])
  const lngInt = parseInt(cols[3])
  if (!nameAz || isNaN(latInt) || isNaN(lngInt)) continue
  const lat = latInt / 1_000_000
  const lng = lngInt / 1_000_000
  if (!exitsByStation.has(nameAz)) exitsByStation.set(nameAz, [])
  exitsByStation.get(nameAz).push({ lat, lng })
}

// --- Parse passenger_2025.csv ---
const passengerRaw = readFileSync(resolve(root, 'datasets/passenger_2025.csv'), 'utf-8')
const passengerLines = passengerRaw.split('\n').slice(1).filter(l => l.trim())

// Accumulate ridership per station name
const ridershipSum = new Map()   // name → total
const ridershipCount = new Map() // name → day count
for (const line of passengerLines) {
  const cols = line.split(';').map(c => c.replace(/^"|"$/g, '').trim())
  const stationName = cols[1]
  const count = parseInt(cols[2])
  if (!stationName || isNaN(count)) continue
  ridershipSum.set(stationName, (ridershipSum.get(stationName) ?? 0) + count)
  ridershipCount.set(stationName, (ridershipCount.get(stationName) ?? 0) + 1)
}

// Average ridership per station
const avgRidership = new Map()
for (const [name, total] of ridershipSum) {
  avgRidership.set(name, Math.round(total / ridershipCount.get(name)))
}

// --- Name fixups: pathways.csv nameAz → passenger_2025.csv name ---
const NAME_FIXUPS = new Map([
  ['28.05.2026 00:00:00', '28 May'],
  ['Şah İsmayıl Xətai', 'Xətai'],
  ['Memar Əcəmi 2', 'Memar Əcəmi-2'],
  ['Koroğlu', 'Koroglu'],
])

// --- Build metro stations array ---
const stations = []
for (const [nameAz, exits] of exitsByStation) {
  const passengerKey = NAME_FIXUPS.get(nameAz) ?? nameAz
  const ridership = avgRidership.get(passengerKey) ?? 0
  stations.push({ nameAz, exits, dailyRidership: ridership })
}

// Sort by nameAz for stable output
stations.sort((a, b) => a.nameAz.localeCompare(b.nameAz, 'az'))

// --- Write lib/metro-stations.ts ---
const tsLines = [
  '// AUTO-GENERATED by scripts/build-metro-data.mjs — do not edit manually',
  '// Source: datasets/pathways.csv + datasets/passenger_2025.csv',
  'import { haversineMetres } from \'./geo\'',
  '',
  'export interface MetroStation {',
  '  nameAz: string',
  '  exits: Array<{ lat: number; lng: number }>',
  '  dailyRidership: number',
  '}',
  '',
  'export const METRO_STATIONS: MetroStation[] = ' + JSON.stringify(stations, null, 2),
  '',
  '/** Returns nearest metro exit distance + station ridership, or null if >2 000 m away. */',
  'export function getNearestMetro(',
  '  lat: number,',
  '  lng: number,',
  '): { distance: number; ridership: number } | null {',
  '  let best: { distance: number; ridership: number } | null = null',
  '  for (const station of METRO_STATIONS) {',
  '    for (const exit of station.exits) {',
  '      const d = haversineMetres(lat, lng, exit.lat, exit.lng)',
  '      if (!best || d < best.distance) {',
  '        best = { distance: Math.round(d), ridership: station.dailyRidership }',
  '      }',
  '    }',
  '  }',
  '  if (!best || best.distance > 2000) return null',
  '  return best',
  '}',
]
writeFileSync(resolve(root, 'lib/metro-stations.ts'), tsLines.join('\n'), 'utf-8')
console.log(`Written lib/metro-stations.ts — ${stations.length} stations`)
for (const s of stations) {
  console.log(`  ${s.nameAz}: ${s.dailyRidership}/day avg, ${s.exits.length} exits`)
}
```

## Task 3: Build script + `lib/settlements.ts`

**Files:**
- Create: `scripts/build-settlements.mjs`
- Create: `lib/settlements.ts` (generated by running the script)

The script reads the GeoJSON (features with `TYPE` and `TYPE_AZ` fields and Point coordinates in [lng, lat] order), maps each feature to an urban tier, and writes a compact TypeScript array.

**Tier mapping rules:**
- `TYPE_AZ` is `"şəhər"` AND (`NAME_AZ` contains `"BAKI"` or `"SUMQAYIT"` or `"Bakı"` or `"Sumqayıt"`) → `'metro-city'`
- `TYPE_AZ` is `"şəhər"` → `'city'`
- `TYPE_AZ` is `"qəsəbə"` → `'town'`
- Everything else (kənd, etc.) → `'rural'`

- [ ] **Step 1: Create `scripts/build-settlements.mjs`**

```js
// scripts/build-settlements.mjs
// Run once: node scripts/build-settlements.mjs
import { readFileSync, writeFileSync } from 'fs'
import { resolve, dirname } from 'path'
import { fileURLToPath } from 'url'

const __dir = dirname(fileURLToPath(import.meta.url))
const root = resolve(__dir, '..')

const METRO_CITIES = new Set(['BAKI', 'SUMQAYIT', 'Bakı', 'Sumqayıt', 'SUMQAYIT'])

function getTier(nameAz, typeAz) {
  if (typeAz === 'şəhər') {
    const upper = nameAz.toUpperCase()
    if (upper.includes('BAK') || upper.includes('SUMQ')) return 'metro-city'
    return 'city'
  }
  if (typeAz === 'qəsəbə') return 'town'
  return 'rural'
}

console.log('Reading GeoJSON...')
const raw = readFileSync(
  resolve(root, 'datasets/azrbaycann_hrlri_v_inzibati_razi_vahidlri.geojson'),
  'utf-8'
)
const geojson = JSON.parse(raw)
console.log(`Loaded ${geojson.features.length} features`)

const settlements = []
for (const feature of geojson.features) {
  const coords = feature.geometry?.coordinates
  const props = feature.properties
  if (!coords || !props) continue
  const [lng, lat] = coords
  const nameAz = props.NAME_AZ || ''
  const typeAz = props.TYPE_AZ || ''
  const tier = getTier(nameAz, typeAz)
  settlements.push({ lat: +lat.toFixed(6), lng: +lng.toFixed(6), tier })
}

const tierCounts = settlements.reduce((acc, s) => {
  acc[s.tier] = (acc[s.tier] ?? 0) + 1; return acc
}, {})
console.log('Tier distribution:', tierCounts)

const tsLines = [
  '// AUTO-GENERATED by scripts/build-settlements.mjs — do not edit manually',
  '// Source: datasets/azrbaycann_hrlri_v_inzibati_razi_vahidlri.geojson',
  'import { haversineMetres } from \'./geo\'',
  '',
  'export type UrbanTier = \'metro-city\' | \'city\' | \'town\' | \'rural\'',
  '',
  'interface Settlement { lat: number; lng: number; tier: UrbanTier }',
  '',
  'const SETTLEMENTS: Settlement[] = ' + JSON.stringify(settlements),
  '',
  '/** Returns the urban tier of the nearest settlement to the given coordinates. */',
  'export function getUrbanTier(lat: number, lng: number): UrbanTier {',
  '  let best: { distance: number; tier: UrbanTier } | null = null',
  '  for (const s of SETTLEMENTS) {',
  '    const d = haversineMetres(lat, lng, s.lat, s.lng)',
  '    if (!best || d < best.distance) best = { distance: d, tier: s.tier }',
  '  }',
  '  return best?.tier ?? \'rural\'',
  '}',
]

writeFileSync(resolve(root, 'lib/settlements.ts'), tsLines.join('\n'), 'utf-8')
console.log(`Written lib/settlements.ts — ${settlements.length} settlements`)
```

```ts
async function fetchFromOverpass(query: string): Promise<{ elements: OSMElement[] }> {
  let lastError: Error = new Error('Overpass unavailable')
  for (const url of OVERPASS_ENDPOINTS) {
    const controller = new AbortController()
    const timeoutId = setTimeout(() => controller.abort(), 15000)
    try {
      const response = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: `data=${encodeURIComponent(query)}`,
        signal: controller.signal,
      })
      clearTimeout(timeoutId)
      if (!response.ok) {
        lastError = new Error(`Overpass API error: ${response.status}`)
        continue
      }
      return (await response.json()) as { elements: OSMElement[] }
    } catch (err) {
      clearTimeout(timeoutId)
      lastError = err as Error
    }
  }
  throw lastError
}

function inferAreaType(elements: OSMElement[]): 'residential' | 'commercial' | 'mixed' {

function resolveOSMTags(businessType: string): string[] | null {
  const lower = businessType.toLowerCase()
  for (const [keywords, osmValues] of COMPETITOR_ALIASES) {
    if (keywords.some((k) => lower.includes(k))) return osmValues
  }
  return null

/** Returns element centre coordinates if available (node: lat/lon, way: center). */
function elementCoords(e: OSMElement): { lat: number; lng: number } | null {
  if (e.lat !== undefined && e.lon !== undefined) return { lat: e.lat, lng: e.lon }
  if (e.center) return { lat: e.center.lat, lng: e.center.lon }
  return null
}

/**
 * Distance-weighted competitor count.
 * < 200 m: weight 1.0   |   200–500 m: weight 0.5   |   no coords: weight 0.5
 */
function countCompetitors(
  elements: OSMElement[],
  businessType: string,
  pinLat: number,
  pinLng: number
): number {
  const lower = businessType.toLowerCase()
  const osmTags = resolveOSMTags(businessType)

  let weighted = 0
  for (const e of elements) {
    const name = (e.tags?.name || '').toLowerCase()
    const amenity = e.tags?.amenity || ''
    const shop = e.tags?.shop || ''
    const leisure = e.tags?.leisure || ''

    const isMatch =
      name.includes(lower) ||
      (osmTags
        ? osmTags.includes(amenity) || osmTags.includes(shop) || osmTags.includes(leisure)
        : amenity.toLowerCase().includes(lower) ||
          shop.toLowerCase().includes(lower) ||
          leisure.toLowerCase().includes(lower))

    if (!isMatch) continue

    const coords = elementCoords(e)
    if (coords) {
      const dist = haversineMetres(pinLat, pinLng, coords.lat, coords.lng)
      weighted += dist < 200 ? 1 : 0.5
    } else {
      weighted += 0.5
    }
  }
  return Math.round(weighted)
}

const BAD_LAND_USES = new Set([
  'cemetery', 'grave_yard', 'military', 'industrial',
  'landfill', 'quarry', 'construction', 'prison',
])

function extractLandUse(elements: OSMElement[]): string | null {
  for (const e of elements) {
    const lu = e.tags?.landuse || e.tags?.amenity || e.tags?.leisure || ''
    if (BAD_LAND_USES.has(lu)) return lu
  }
  return null
}

export async function fetchPlacesContext(
  lat: number,
  lng: number,
  businessType: string
): Promise<PlacesContext> {
  const [businessData, landUseData] = await Promise.all([
    fetchFromOverpass(buildQuery(lat, lng)).catch(() => ({ elements: [] as OSMElement[] })),
    fetchFromOverpass(buildLandUseQuery(lat, lng)).catch(() => ({ elements: [] as OSMElement[] })),
  ])

  const allElements = businessData.elements || []
  const { businessElements, roadElements } = splitElements(allElements)
  const landUse = extractLandUse(landUseData.elements || [])

  const azCount = countAzCompetitors(lat, lng, businessType)
  const osmCount = countCompetitors(businessElements, businessType, lat, lng)
  const competitors = azCount >= 0 ? Math.max(azCount, osmCount) : osmCount
  const recognized = resolveOSMTags(businessType) !== null || azCount >= 0

  const metro = getNearestMetro(lat, lng)
  const urbanTier = getUrbanTier(lat, lng)

  return {
    competitors,
    areaType: inferAreaType(businessElements),
    amenities: extractAmenities(businessElements),
    totalBusinesses: businessElements.length,
    landUse,
    recognized,
    busStops: extractBusStops(businessElements),
    parking: extractParking(businessElements),
    groceryStores: extractGroceryStores(businessElements),
    majorRoads: roadElements.length,
    metroDistance: metro?.distance ?? null,
    metroRidership: metro?.ridership ?? null,
    urbanTier,
  }
}
```

```ts
import type { PlacesContext, FactorResult } from './types'

export interface ScoreResult {
  score: number
  factors: FactorResult[]
}

const LAND_USE_CAPS: Record<string, number> = {
  cemetery: 8,
  grave_yard: 8,
  military: 8,
  landfill: 5,
  quarry: 10,
  prison: 8,
  industrial: 30,
  construction: 25,
}

function metroTierScore(ridership: number | null): number {
  if (ridership === null) return 0
  if (ridership >= 30000) return 12
  if (ridership >= 20000) return 10
  if (ridership >= 10000) return 7
  if (ridership >= 5000) return 4
  return 2
}

function roadScore(majorRoads: number): number {
  return Math.min(8, Math.round(Math.min(majorRoads, 3) * (8 / 3)))
}

function busStopScore(busStops: number): number {
  if (busStops === 0) return 0
  if (busStops === 1) return 3
  if (busStops === 2) return 5
  return 7
}

function groceryScore(groceryStores: number): number {
  if (groceryStores === 0) return 0
  if (groceryStores === 1) return 2
  if (groceryStores === 2) return 4
  return 5
}

function densityScore(totalBusinesses: number): number {
  if (totalBusinesses >= 50) return 10
  if (totalBusinesses >= 25) return 7
  if (totalBusinesses >= 10) return 4
  return 0
}

export function calculateScore(ctx: PlacesContext): ScoreResult {
  const competitionScore = Math.max(0, Math.round(22 - ctx.competitors * 2.2))
  const footTrafficScore = Math.min(20, metroTierScore(ctx.metroRidership) + roadScore(ctx.majorRoads))
  const areaScore = ctx.areaType === 'commercial' ? 13 : ctx.areaType === 'mixed' ? 9 : 5
  const urbanScore =
    ctx.urbanTier === 'metro-city' ? 10
    : ctx.urbanTier === 'city' ? 7
    : ctx.urbanTier === 'town' ? 4
    : 1
  const accessibilityScore = busStopScore(ctx.busStops) + (ctx.parking > 0 ? 5 : 0)
  const nearbyServicesScore = groceryScore(ctx.groceryStores) + Math.min(3, ctx.amenities.length > 1 ? 3 : ctx.amenities.length)
  const businessDensityScore = densityScore(ctx.totalBusinesses)

  const raw = Math.round(
    competitionScore + footTrafficScore + areaScore + urbanScore +
    accessibilityScore + nearbyServicesScore + businessDensityScore
  )

  const cap =
    ctx.landUse && LAND_USE_CAPS[ctx.landUse] !== undefined
      ? LAND_USE_CAPS[ctx.landUse]
      : 95

  const score = Math.min(raw, cap)

  const factors: FactorResult[] = [
    { key: 'competition', score: Math.min(competitionScore, 22), max: 22 },
    { key: 'footTraffic', score: Math.min(footTrafficScore, 20), max: 20 },
    { key: 'areaType', score: areaScore, max: 13 },
    { key: 'urbanTier', score: urbanScore, max: 10 },
    { key: 'accessibility', score: Math.min(accessibilityScore, 12), max: 12 },
    { key: 'nearbyServices', score: Math.min(nearbyServicesScore, 8), max: 8 },
    { key: 'businessDensity', score: businessDensityScore, max: 10 },
  ]

  return { score, factors }
}
```

function parseResponse(content: string, score: number): AnalysisResult {
  const cleaned = content
    .replace(/```json\n?/g, '')
    .replace(/```\n?/g, '')
    .trim()
  const parsed = JSON.parse(cleaned) as Omit<AnalysisResult, 'score' | 'factors'>
  if (
    typeof parsed.summary !== 'string' ||
    typeof parsed.detail !== 'string' ||
    !Array.isArray(parsed.pros) ||
    !Array.isArray(parsed.cons) ||
    typeof parsed.verdict !== 'string'
  ) {
    throw new Error('Invalid response shape')
  }
  return { score, summary: parsed.summary, detail: parsed.detail, pros: parsed.pros, cons: parsed.cons, verdict: parsed.verdict }
}

export async function analyzeLocation(
  lat: number,
  lng: number,
  businessType: string,
  ctx: PlacesContext,
  score: number
): Promise<AnalysisResult> {
  if (!process.env.GROQ_API_KEY) {
    throw new Error('GROQ_API_KEY environment variable is not set')
  }
  const client = new Groq({ apiKey: process.env.GROQ_API_KEY })
  const prompt = buildPrompt(lat, lng, businessType, ctx, score)

  for (let attempt = 0; attempt < 2; attempt++) {
    const response = await client.chat.completions.create({
      model: MODEL,
      messages: [{ role: 'user', content: prompt }],
    })
    const content = response.choices[0]?.message?.content || ''
    try {
      return parseResponse(content, score)
    } catch {
      if (attempt === 1) throw new Error('Groq returned invalid JSON after 2 attempts')
    }
  }
  throw new Error('Groq returned invalid JSON after 2 attempts')
}
